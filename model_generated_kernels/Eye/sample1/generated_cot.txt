现在，如何在Ascend C中实现这个算子：

1. **Tiling策略**：
   - 总数据长度为num_rows × num_columns = 18×10=180个元素。
   - 每个AI Core处理的批次数需要计算。GetBlockNum()获取AI Core总数（假设为48个AIV）。总批次为48×32=1,536。因此每个核处理1,536 / 48=32个批次。
   - 每个核需要处理32个矩阵，每个矩阵大小为180元素。数据将被分片以适应UB缓冲区。

2. **Kernel实现**：
   - 对于每个批次，生成一个填充了0的矩阵，然后在对角线上设置1。
   - 使用DataCopy将整个零矩阵写入GM，然后使用矢量指令将对角线元素设置为1.0。

3. **流水线设计**：
   - 由于输出是显式的（不需要输入数据），流程将直接生成数据。
   - 每个批次将被处理为一个独立的矩阵。每个矩阵可以被分片到UB中。

4. **数据对齐**：
   - float32是32位，因此每个元素4字节。确保所有内存访问32字节对齐。
   - 例如，16个元素（64字节）是32字节对齐的。

5. **对角线计算**：
   - 对于每个矩阵，索引i处的对角线元素在范围[min(num_rows, num_columns)]内。对于行i，列i上元素的位置为i × num_columns + i（如果i < min）。

现在，逐步实现代码：

首先，定义TilingData结构以传递参数。需要将num_rows、num_columns、total_batches等传递给kernel。在Host侧，将计算每个核处理的批次范围。

在kernel侧：
- 初始化GlobalTensor为当前核的批次范围。
- 对于每个批次，分配UB缓冲区。
   - 填充零。
   - 将对角线元素设置为1.0。
   - 将整个矩阵复制回GM。

对于填充零，可以使用Duplicate指令。设置对角线可以使用矢量赋值。

现在，考虑分块。由于矩阵较小（18×10=180元素），可以将整个矩阵加载到UB中。每个矩阵180×4=720字节。UB大小通常为256KB，因此双缓冲可行。

代码结构如下：

```cpp
class KernelEye {
public:
    __aicore__ inline void Init(...) {
        // 设置GM缓冲区并初始化队列
        pipe.InitBuffer(outQueue, BUFFER_NUM, tileLength * sizeof(float));
    }

    __aicore__ inline void Process() {
        // 对每个批次循环
        for每个批次:
            CopyOut(生成零矩阵并设置对角线)
    }

private:
    __aicore__ inline void CopyOut(int progress) {
        LocalTensor<float> zLocal = outQueue.AllocTensor<float>();
        // 填充零
        AscendC::Duplicate(zLocal, 0.0f, tileLength);
        // 设置对角线
        for i in 0..min(num_rows, num_columns)-1:
            zLocal[i * num_columns + i] = 1.0f;
        // 复制回GM
        DataCopy(gm[progress * tileLength], zLocal, tileLength);
        outQueue.FreeTensor(zLocal);
    }
};
```

#include "kernel_operator.h"

using namespace AscendC;

typedef struct {
    uint32_t num_rows;
    uint32_t num_columns;
    uint32_t min_diag;
    uint32_t totalLength;
    uint32_t tileNum;
} TilingDataDef;

constexpr int32_t BUFFER_NUM = 2;

class KernelEye {
public:
    __aicore__ inline KernelEye() {}
    
    __aicore__ inline void Init(GM_ADDR y, GM_ADDR y_ref, uint32_t num_rows, uint32_t num_columns, 
                              uint32_t min_diag, uint32_t tileNum)
    {
        this->num_rows = num_rows;
        this->num_columns = num_columns;
        this->min_diag = min_diag;
        this->tileNum = tileNum;
        this->tileLength = num_rows * num_columns;

        // 每个核处理32个批次（48*32=1536总条目）
        uint32_t batch_stride = num_rows * num_columns;
        
        yGm.SetGlobalBuffer((__gm__ float*)y + GetBlockIdx() * 32 * batch_stride, 32 * batch_stride);
        pipe.InitBuffer(outQueue, BUFFER_NUM, tileLength * sizeof(float));
    }

    __aicore__ inline void Process()
    {
        for (int32_t i = 0; i < 32; i++) {
            GenerateMatrix(i);
        }
    }

private:
    __aicore__ inline void GenerateMatrix(int32_t batch_idx)
    {
        LocalTensor<float> zLocal = outQueue.AllocTensor<float>();
        
        // 初始化为0
        AscendC::Duplicate(zLocal, 0.0f, tileLength);
        
        // 设置对角线为1
        for (uint32_t i = 0; i < min_diag; i++) {
            zLocal.SetValue(i * num_columns + i, 1.0f);
        }
        
        outQueue.EnQue(zLocal);
        
        // 数据搬运到GM
        LocalTensor<float> out_tensor = outQueue.DeQue<float>();
        AscendC::DataCopy(yGm[batch_idx * tileLength], out_tensor, tileLength);
        outQueue.FreeTensor(out_tensor);
    }

private:
    AscendC::TPipe pipe;
    AscendC::TQue<AscendC::QuePosition::VECOUT, BUFFER_NUM> outQueue;
    AscendC::GlobalTensor<float> yGm;
    
    uint32_t num_rows;
    uint32_t num_columns;
    uint32_t min_diag;
    uint32_t tileNum;
    uint32_t tileLength;
};

extern "C" __global__ __aicore__ void eye(GM_ADDR y, GM_ADDR y_ref, GM_ADDR workspace) 
{
    TilingDataDef tiling_data = {
        .num_rows = 18,
        .num_columns = 10,
        .min_diag = 10,
        .totalLength = 180,
        .tileNum = 1
    };
    
    KernelEye op;
    op.Init(y, y_ref, tiling_data.num_rows, tiling_data.num_columns, 
            tiling_data.min_diag, tiling_data.tileNum);
    op.Process();
}
